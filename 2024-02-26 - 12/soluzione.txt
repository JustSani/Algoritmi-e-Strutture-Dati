int trovaParola(int riga,int colonna,char ** testo,char ** elenco,int np){
	int i,j, countChar;	
	// scorriamo tutte le parole
	for(i = 0; i < np; i++){
		// controlliamo se la parola corrente è uguale a quella in questa posizione
		countChar = 0;
		for(j = 0; j < strlen(elenco[i]) -1 && colonna + j < NC; j++){
			if(elenco[i][j] == testo[riga][colonna + j])
				countChar ++;
		}
		
		if(countChar == strlen(elenco[i] - 1)
			return i;
	}

	return -1;

}

		return temp;
	}
	if(cmpNode(a->right,z) == 0)
		return rExtractLast(a->left, z);

int checkSum(int sol,int B,int minS){
